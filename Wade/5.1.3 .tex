\begin{exercise}
Let $E=\{1/n: n\in \N\}$. Prove that the function 
\[
f(x)=
\begin{cases}
    1 & \quad x \in E \\
    0 & \quad \text{otherwise}
\end{cases}
\]
is integrable on $[0,1]$. What is the value of $\int_{0}^{1} f(x)\,dx$?
\end{exercise} 
\begin{proof}
Let $a,b \in [0,1]$ with $a<b$. Then there is a real number $\xi \in [a,b] \setminus \Q$. 
Thus, we have $\inf_{x \in [a,b]} f(x)=0$, and it follows that $L(f,P)=0$ if $P$ is any partition of $[0,1]$. 

Given $\varepsilon>0$, use the Archimedean Principle to choose an $n \in \N$ such that $2<n\varepsilon$ with $n>1$. 
Then define 
\[
E_k \coloneqq \left\{|x_k-y_k|:\frac{1}{n-k+1}<x_k<y_k<\frac{1}{n-k}\right\}
\]
for $k=1,2,\ldots,n-1$. 
Since $\sup E_k = \frac{1}{n-k} - \frac{1}{n-k+1}$ for $1 \leq k \leq n-1$, there exist points $x_k, y_k \in \left(\frac{1}{n-k+1},\frac{1}{n-k}\right)$ such that
\[
\frac{1}{n-k} - \frac{1}{n-k+1} - \frac{\varepsilon}{2(n-1)}<|x_k-y_k|
\]
with $x_k<y_k$ for each $1 \leq k \leq n-1$.

Now, set $P \coloneqq \left \{0, \frac{1}{n}, x_1, y_1, x_2, y_2, \ldots , x_{n-1}, y_{n-1}, 1 \right \}$, then $P$ is a partition of $[0,1]$. 
Thus,
\begin{align*}
U(f,P)-L(f,P) &= 1 - \sum_{k=1}^{n-1} |x_k-y_k| \\
              &< 1 - \sum_{k=1}^{n-1} \left (\frac{1}{n-k} - \frac{1}{n-k+1} - \frac{\varepsilon}{2(n-1)}\right) \\
              &= 1 - \left(1 - \frac{1}{n} - \frac{\varepsilon}{2}\right) \\
              &= \frac{1}{n} + \frac{\varepsilon}{2} < \frac{\varepsilon}{2} + \frac{\varepsilon}{2} = \varepsilon.
\end{align*}
Therefore, $f$ is integrable on $[0,1]$.
Hence, we obtain $\int_{0}^{1} f(x) \,dx = \sup L(f,P) = 0$.
\end{proof}
