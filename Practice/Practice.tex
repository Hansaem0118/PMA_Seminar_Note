\begin{theorem}
Suppose that $\{x_n\}$ and $\{y_n\}$ are real sequences. If $\{x_n\}$ and $\{y_n\}$ are convergent, then

\[
\lim_{n \to \infty} (x_n + y_n)
=
\lim_{n \to \infty} x_n
+
\lim_{n \to \infty} y_n.
\]

\end{theorem}

\begin{proof}
Let $L=\lim_{n \to \infty} x_n$ and $M=\lim_{n \to \infty} y_n$. Given $\varepsilon >0$, there exist $N_1,N_2 \in \mathbb{N}$ such that
\[
 n \geq N_1 \quad \text{implies} \quad |x_n-L| < \varepsilon/2
 \]and 
 \[
 n \geq N_2 \quad \text{implies} \quad |y_n-M| < \varepsilon/2.
 \]
 Take $N=\max{\{N_1, N_2\}}$, then $ n \geq N$ implies
 \begin{align*}
 |(x_n+y_n)-(L+M)|&=|(x_n-L)+(y_n-M)|\\ 
 &\leq |x_n-L|+|y_n-M|\\
 &<\varepsilon/2+\varepsilon/2=\varepsilon.
 \end{align*}
\end{proof}
